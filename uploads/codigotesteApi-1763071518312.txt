        // Vari√°veis globais
        let jwtToken = localStorage.getItem('jwtToken') || null;
        let ws; // Cliente Socket.IO

        // --- Fun√ß√µes de Utilit√°rio e JWT ---

        function updateJwtDisplay() {
            const display = document.getElementById('jwt-display');
            if (jwtToken) {
                display.textContent = jwtToken.substring(0, 40) + '... (Armazenado)';
                display.style.backgroundColor = '#d4edda'; 
                display.style.borderColor = '#c3e6cb';
            } else {
                display.textContent = 'Nenhum JWT armazenado.';
                display.style.backgroundColor = '#ffeeba'; 
                display.style.borderColor = '#ffc107';
            }
        }

        function saveJWT(token) {
            jwtToken = token;
            localStorage.setItem('jwtToken', token);
            updateJwtDisplay();
        }

        function clearJWT() {
            jwtToken = null;
            localStorage.removeItem('jwtToken');
            updateJwtDisplay();
        }

        async function loginAndSaveToken() {
            const email = document.getElementById('login-email').value;
            const senha = document.getElementById('login-senha').value;
            const body = { email, senha };
            const responseDiv = document.getElementById('response-token');

            try {
                const apiBase = document.getElementById('api-base-url').value;
                // ROTA CORRIGIDA: /Auth/
                const url = `${apiBase}/Auth/`; 

                responseDiv.textContent = "Enviando requisi√ß√£o...";

                const response = await fetch(url, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(body)
                });
                
                let text = "";
                let data = null;
                try {
                    text = await response.text();
                    if (text) data = JSON.parse(text);
                } catch (e) {
                    data = { message: "Resposta n√£o √© JSON (veja o 'raw').", raw: text };
                }

                if (response.ok && data.token) {
                    saveJWT(data.token);
                    responseDiv.textContent = `Status: ${response.status}\nToken Salvo com Sucesso!`;
                    responseDiv.style.color = 'green';
                } else {
                    responseDiv.textContent = `Status: ${response.status}\n${JSON.stringify(data, null, 2)}`;
                    responseDiv.style.color = 'red';
                }
            } catch (error) {
                responseDiv.textContent = `Erro de Conex√£o: ${error.message}`;
                responseDiv.style.color = 'red';
            }
        }

        // --- Fun√ß√µes de Requisi√ß√£o HTTP ---

        async function sendRequest(method, path, body, responseDivId) {
            const responseDiv = document.getElementById(responseDivId);
            const apiBase = document.getElementById('api-base-url').value;
            const url = `${apiBase}${path}`;
            
            responseDiv.textContent = "Enviando requisi√ß√£o...";
            responseDiv.style.color = '#000';

            // ROTAS CORRIGIDAS NA VERIFICA√á√ÉO
            const isProtected = !['/Usuarios/', '/Auth/'].includes(path); 
            
            if (isProtected && !jwtToken) {
                responseDiv.textContent = "ERRO: Rota protegida. Fa√ßa login primeiro para obter o JWT.";
                responseDiv.style.color = 'red';
                return;
            }

            const headers = {
                'Content-Type': 'application/json',
            };
            if (isProtected) {
                headers['Authorization'] = `Bearer ${jwtToken}`;
            }

            let text = ""; 

            try {
                const fetchOptions = {
                    method: method, // Agora aceita 'POST', 'GET', 'PUT', etc.
                    headers: headers,
                };

                if (body && (method === 'POST' || method === 'PUT')) {
                    fetchOptions.body = JSON.stringify(body);
                }

                const response = await fetch(url, fetchOptions);
                let data = null;

                try {
                    text = await response.text();
                    if (text) {
                        data = JSON.parse(text); 
                    }
                } catch (e) {
                    data = { message: "Resposta n√£o √© JSON (veja o 'raw').", raw: text };
                }

                responseDiv.textContent = `Status: ${response.status} (${response.statusText})\n\n${JSON.stringify(data || {}, null, 2)}`;
                responseDiv.style.color = response.ok ? 'green' : 'red';
            
            } catch (error) { 
                responseDiv.textContent = `Erro de Conex√£o: ${error.message}`;
                responseDiv.style.color = 'red';
            }
        }

        // --- Fun√ß√µes de Visualiza√ß√£o de Cartas ---

        // Armazena as m√£os de todos os jogadores para evitar que o display seja limpo
        // Estrutura: { [par_id]: [card1, card2, ...] }
        let allPlayerHands = {};
        const DEALER_ID = 'DEALER';

        function clearCardDisplay() {
            document.getElementById('player-cards-display').innerHTML = 'Aguardando cartas...';
            document.getElementById('dealer-cards-display').innerHTML = 'Aguardando cartas...';
            allPlayerHands = {}; // Limpa o estado
            logWS("Display de Cartas Limpo.", 'INFO');
        }

        /**
         * Gera o HTML de um conjunto de cartas.
         * @param {Array} cardsArray - Array de objetos de carta.
         * @param {string} parId - O ID do participante (ou DEALER).
         * @returns {string} O HTML das cartas.
         */
        function createCardsHtml(cardsArray) {
            let cardsHtml = '';
            cardsArray.forEach(card => {
                if (!card || !card.image || !card.code) return; 

                const value = card.value || card.code;
                const suit = card.suit || '';
                const altText = `${value} de ${suit}`;
                
                const textColor = suit.includes('DIAMONDS') || suit.includes('HEARTS') ? 'red' : 'black';

                cardsHtml += `
                    <div style="border: 1px solid #000; padding: 5px; background-color: white; border-radius: 3px; font-size: 0.8em; text-align: center; width: 60px; box-shadow: 2px 2px 5px rgba(0,0,0,0.2);">
                        <img src="${card.image}" alt="${altText}" style="width: 100%; height: auto; display: block; margin: auto; border-radius: 2px;"><br>
                        <span style="display: block; margin-top: 3px; font-weight: bold; color: ${textColor};">${value}</span>
                    </div>
                `;
            });
            return cardsHtml;
        }


        /**
         * Renderiza as cartas, agrupando por participante, se for o display do jogador.
         * @param {Object|Array} cards - A carta (objeto) ou cartas (array) recebidas.
         * @param {string} targetId - O ID do elemento DIV onde as cartas ser√£o exibidas.
         * @param {number|string} [parId=null] - O ID do participante.
         */
        function renderCards(cards, targetId, parId = null) {
            const cardsArray = Array.isArray(cards) ? cards : (cards ? [cards] : []);
            
            if (targetId === 'dealer-cards-display') {
                const display = document.getElementById(targetId);
                display.innerHTML = createCardsHtml(cardsArray); // Sempre substitui a m√£o do dealer
                return;
            }

            // L√≥gica para Jogadores (targetId === 'player-cards-display')

            if (cardsArray.length === 0 || parId === null) {
                 // Nenhuma carta ou ID para processar
                 return;
            }

            // 1. Atualiza o estado da m√£o do jogador
            const playerId = `par-${parId}`;
            
            // Inicializa ou limpa a m√£o se estiver recebendo a m√£o inicial (2 cartas)
            // Ou se o estado ainda n√£o existe (primeira carta recebida).
            if (!allPlayerHands[playerId] || cardsArray.length > 1) {
                allPlayerHands[playerId] = cardsArray;
            } else {
                // Adiciona cartas (para hit/compra de carta √∫nica)
                allPlayerHands[playerId] = allPlayerHands[playerId].concat(cardsArray);
            }

            // 2. Renderiza TODAS as m√£os do estado para o display
            const display = document.getElementById('player-cards-display');
            display.innerHTML = ''; // Limpa o display para reconstruir
            
            let allHandsHtml = '';
            
            // Verifica se o display ainda est√° com o texto inicial e o substitui
            if (Object.keys(allPlayerHands).length === 0) {
                 display.innerHTML = 'Aguardando cartas...';
            }


            for (const id in allPlayerHands) {
                const currentParId = id.replace('par-', '');
                const cardsInHand = allPlayerHands[id];

                allHandsHtml += `
                    <div class="player-hand-group">
                        <p style="font-weight: bold; margin-bottom: 5px; margin-top: 0; color: #2c3e50;">
                            Jogador (Par ID: ${currentParId})
                        </p>
                        <div class="cards-container">
                            ${createCardsHtml(cardsInHand)}
                        </div>
                    </div>
                `;
            }
            
            display.innerHTML = allHandsHtml;
        }


        // --- Fun√ß√µes de WebSocket (Socket.IO) ---

        function logWS(message, type = 'INFO') {
            const logBox = document.getElementById('ws-log');
            const now = new Date().toLocaleTimeString();
            let color = 'inherit';
            switch(type) {
                case 'ERROR': color = '#ff4d4d'; break;
                case 'EVENT': color = '#4da6ff'; break;
                case 'SUCCESS': color = '#00cc00'; break;
                case 'WARN': color = '#ffc107'; break;
            }
            logBox.innerHTML += `<span style="color: ${color};">[${now} - ${type}]</span> ${message}\n`;
            logBox.scrollTop = logBox.scrollHeight;
        }

        function connectWebSocket() {
            const wsUrl = document.getElementById('ws-url').value;
            const connectBtn = document.getElementById('ws-connect-btn');
            
            if (ws && ws.connected) {
                ws.disconnect();
                logWS("Desconectando socket anterior...", 'WARN');
            }
            
            logWS(`Tentando conectar a ${wsUrl} (via Socket.IO)...`);
            connectBtn.style.backgroundColor = 'orange';
            connectBtn.textContent = 'Conectando...';

            ws = io(wsUrl, { transports: ['websocket'] }); 

            ws.on("connect", () => {
                logWS("Conectado com sucesso via Socket.IO!", 'SUCCESS');
                connectBtn.style.backgroundColor = 'green';
                connectBtn.textContent = 'Conectado (clique para reconectar)';
            });

            ws.on("disconnect", () => {
                logWS("Conex√£o fechada. (Tente reconectar)", 'WARN');
                connectBtn.style.backgroundColor = '#dc3545';
                connectBtn.textContent = 'Desconectado (clique para conectar)';
            });

            ws.on("connect_error", (error) => {
                logWS(`ERRO DE CONEX√ÉO: ${error.message || 'Verifique a URL, o CORS e o servidor'}`, 'ERROR');
                connectBtn.style.backgroundColor = '#dc3545';
                connectBtn.textContent = 'Erro (clique para tentar novamente)';
            });

            // --- Eventos de Jogo ---
            ws.on("mensagem", (data) => {
                logWS(`[mensagem]: ${data}`, 'EVENT');
            });
            ws.on("ID", (data) => {
                logWS(`[ID]: Meu Socket ID √©: ${data}`, 'EVENT');
            });
            ws.on("Atualizar_Salas", (data) => {
                logWS(`[Atualizar_Salas]: ${JSON.stringify(data)}`, 'EVENT');
            });
            ws.on("Novo_Usuario", (data) => {
                logWS(`[Novo_Usuario]: ${data}`, 'EVENT');
            });
            ws.on("Usuario_saiu", (data) => {
                logWS(`[Usuario_saiu]: ${data}`, 'EVENT');
            });
            ws.on("Iniciar_Rodada", (data) => {
                // Ao iniciar uma rodada, limpa o display de cartas antigo
                clearCardDisplay(); 
                logWS(`[Iniciar_Rodada]: ${data}`, 'EVENT');
            });
            ws.on("Todos_apostaram", (data) => {
                logWS(`[Todos_apostaram]: ${data}`, 'EVENT');
            });
            ws.on("Falta_apostar", (data) => {
                logWS(`[Falta_apostar]: ${data}`, 'EVENT');
            });
            ws.on("Erro_apostar", (data) => {
                logWS(`[Erro_apostar]: ${data}`, 'ERROR');
            });
            
            // Handler para cartasIniciais
            ws.on("cartasIniciais", (cartas, par_id) => {
                logWS(`[cartasIniciais] (P/ ID ${par_id}): ${JSON.stringify(cartas)}`, 'EVENT');
                // Usa renderCards, que agora agrupa as cartas pelo par_id
                renderCards(cartas, 'player-cards-display', par_id);
            });

            // Handler para cartas_dealer
            ws.on("cartas_dealer", (cartas) => {
                logWS(`[cartas_dealer]: ${JSON.stringify(cartas)}`, 'EVENT');
                // Substitui a m√£o do dealer
                renderCards(cartas, 'dealer-cards-display', DEALER_ID);
            });
        }

        function joinSalaWS() {
